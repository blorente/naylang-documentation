\documentclass{standalone}

\usepackage{tikz-uml}
\usetikzlibrary{positioning}

\begin{document}
\begin{tikzpicture}

\umlclass[x=0, y=0]{AbstractVisitor}{}{
	\umlvirt{visit(n : A) : void} \\
	\umlvirt{visit(n : B) : void} \\
	\umlvirt{visit(n : C) : void}
}

\umlclass[x=0, y=-3]{MainVisitor}{}{
	\umlvirt{visit(n : A) override : void} \\
	\umlvirt{visit(n : B) override : void} \\
	\umlvirt{visit(n : C) override : void}
}

\umlclass[x=5.5, y=-3]{ExpandedVisitor}{}{
	\umlvirt{visit(n : A) override : void} \\
	\umlvirt{visit(n : B) override : void} \\
	\umlvirt{visit(n : C) override : void}
}

\umlinherit[geometry=-|]{MainVisitor}{AbstractVisitor}
\umlinherit[geometry=--]{ExpandedVisitor}{AbstractVisitor}
\umlunicompo[arg=super, align2=right, pos=0.8 , angle1=180 , angle2=18 , loopsize=2cm] {ExtensionVisitor}{MainVisitor}

\end{tikzpicture}
\end{document}
