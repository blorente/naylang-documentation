\documentclass{standalone}

\usepackage{tikz-uml}
\usetikzlibrary{positioning}

\begin{document}  
  \def\len{1cm}
  \begin{tikzpicture}
    \begin{umlpackage}{src/core/model/ast}
      \umlsimpleclass[x=-1, y=0]{Statement} {line : int \\ col : int} {line() : int \\ col() : int}{}      
      \begin{umlpackage}[y=-3]{control}
        \umlclass[x=-5, y=0]{Assignment}
          {field : string \\ scope : ExpressionPtr \\ value : ExpressionPtr}
          {field() const : const string \& \\ scope() const : const ExpressionPtr \& \\ value() const :const ExpressionPtr \&}
        \umlclass[x=5, y=0]{IfThen}
          {condition : ExpressionPtr \\ then : vector(StatementPtr)}
          {condition() const : ExpressionPtr \\ thenPart() const :  const vector(StatementPtr) \&}
        \umlclass[x=5, y=-4]{IfThenElse}
          {condition : ExpressionPtr \\ thenp : vector(StatementPtr) \\ elsep : vector(StatementPtr)}
          {condition() const : ExpressionPtr \\ thenPart() const :  const vector(StatementPtr) \& \\ \\ elsePart() const :  const vector(StatementPtr) \&}
        \umlclass[x=-5, y=-4]{While}
          {condition : ExpressionPtr \\ body : vector(StatementPtr)}
          {condition() const : ExpressionPtr \\ body() const :  const vector(StatementPtr) \&}
        \umlemptyclass[x=0.1, y=-2]{Return}
      \end{umlpackage}
    \end{umlpackage}

    % Inheritances
    \umlinherit[geometry=-|]{Assignment}{Statement}
    \umlinherit[geometry=-|]{IfThen}{Statement}
    \umlinherit[geometry=-|]{IfThenElse}{Statement}
    \umlinherit[geometry=-|]{While}{Statement}
    \umlinherit[geometry=-|]{Return}{Statement}

  \end{tikzpicture}
\end{document}
