  \begin{tikzpicture}  
    \begin{umlpackage}[x=0, y=0]{src/core/model/ast}
      \umlclass[x=-1, y=0]{Statement} {line : int \\ col : int} {line() : int \\ col() : int}{}
      \umlclass[x=1, y=0]{GraceAST} {nodes : std::vector<StatementPtr>}{}
      \begin{umlpackage}{expressions}
        \umlclass{Expression} {}{}
        \umlclass{ObjectConstructor} {\_statements : std::vector<StatementPtr>}{}
        \begin{umlpackage}{primitives}
          \umlclass{NumberLiteral} {\_value : double} {value() : double}
          \umlclass{StringLiteral} {\_value : string} {value() : const string \&}
          \umlemptyclass{more}
        \end{umlpackage}{primitives}
        \begin{umlpackage}{requests}
          \umlclass{RequestNode}

      \begin{umlpackage}{control}

        \umlclass{IfThen}
          {\_condition : ExpressionPtr \\ \_then : std::vector<StatementPtr>}
          {condition() const : ExpressionPtr \\ thenPart() const :  const std::vector<StatementPtr> \&}
        \umlemptyclass{more}
      \end{umlpackage}
    \end{umlpackage}
  \end{tikzpicture}
